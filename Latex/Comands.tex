  framerule       = 0.4pt,

% --- Función de costo total (modelo ponderado) ---
% Uso: \costototal{i}{j} genera c_{ij} = d_{ij} + \alpha·a_{ij} + \beta·\delta_{ij}
\newcommand{\costototal}[2]{%
  c_{#1#2} = d_{#1#2} + \alpha \cdot a_{#1#2} + \beta \cdot \delta_{#1#2}%
}
